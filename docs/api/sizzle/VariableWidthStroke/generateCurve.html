Like generate, but the spine follows the quadratic Bézier curve defined by start, control and end. Parameters and cap behaviour are identical to generate; the wobble is applied perpendicular to the curve's local tangent.

static GeneratedStroke generateCurve({
  required Offset start,
  required Offset control,
  required Offset end,
  required Random random,
  double segmentLength = _defaultSegmentLength,
  double minWidth = 3,
  double maxWidth = 26,
  double wobble = 10,
  double widthVariability = 1,
  StrokeEnd? startCap,
  StrokeEnd? endCap,
}) {
  final count = _segmentCount(
    _quadraticLength(start, control, end),
    segmentLength,
  );
  return _strokeAlongSpine(
    spine: _applyWobble(
      _quadraticSamples(start, control, end, count),
      random,
      wobble,
    ),
    random: random,
    minWidth: minWidth,
    maxWidth: maxWidth,
    widthVariability: widthVariability,
    startCap: startCap,
    endCap: endCap,
  );
}
