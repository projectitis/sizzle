Builds a 256x1 RGBA lookup image encoding radius-fraction-vs-position for stops. The value is stored in the red channel and sampled by the shader (which multiplies it by the max radius). The growth mapping is applied here, so the shader stays identical regardless of mode.

Future<Image> buildAmountLut(
  List<HalftoneStop> stops, {
  HalftoneGrowth growth = HalftoneGrowth.linear,
}) {
  const width = 256;
  final pixels = Uint8List(width * 4);
  final sorted = [...stops]..sort((a, b) => a.position.compareTo(b.position));

  for (var x = 0; x < width; x++) {
    final t = x / (width - 1);
    final amount = _amountAt(sorted, t);
    final mapped = switch (growth) {
      HalftoneGrowth.linear => amount,
      HalftoneGrowth.area => math.sqrt(amount),
    };
    final a = (mapped * 255.0).round().clamp(0, 255);
    final o = x * 4;
    pixels[o] = a; // R = radius fraction
    pixels[o + 1] = 0;
    pixels[o + 2] = 0;
    pixels[o + 3] = 255;
  }

  final completer = Completer<Image>();
  decodeImageFromPixels(
    pixels,
    width,
    1,
    PixelFormat.rgba8888,
    completer.complete,
  );
  return completer.future;
}
